üìö Installation & Usage

1. Include Files

<link rel="stylesheet" href="date_picker.css"> <script type="module"> import NepaliDatePicker from './date_picker.js'; </script>

2. HTML Usage (Automatic)

<!-- Nepali Date (BS) --> <input type="npdate" data-mode="BS" data-language="en"> <!-- With Nepali numerals --> <input type="npdate" data-mode="BS" data-language="np"> <!-- Gregorian Date (AD) --> <input type="npdate" data-mode="AD">

3. JavaScript API

import NepaliDatePicker from './date_picker.js'; // Create instance const picker = new NepaliDatePicker('#my-input', { mode: 'BS', language: 'en', onChange: (date, picker) => { console.log('Selected:', date.format('%Y-%m-%d')); } }); // Initialize all npdate inputs NepaliDatePicker.init();

4. Options

{ mode: 'BS', // 'BS' or 'AD' language: 'en', // 'en' or 'np' format: '%Y-%m-%d', // Date format theme: 'auto', // 'auto', 'light', 'dark' position: 'auto', // 'auto', 'top', 'bottom' closeOnSelect: true, // Auto close on select showTodayButton: true, // Show today button showClearButton: true, // Show clear button onChange: null, // Callback function onOpen: null, // Open callback onClose: null // Close callback }
