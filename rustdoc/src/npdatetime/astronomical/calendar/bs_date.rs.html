7/// Represents a date in the astronomical Bikram Sambat calendar
8#[cfg(feature = "serde")]
9use serde::{Deserialize, Serialize};
10
11#[derive(Debug, Clone, Copy, PartialEq, Eq, PartialOrd, Ord)]
12#[cfg_attr(feature = "serde", derive(Serialize, Deserialize))]
13pub struct BsDate {
14    pub year: i32,
15    pub month: u8,
16    pub day: u8,
17}
18
19impl BsDate {
20    /// Creates a new astronomical BS date
21    pub fn new(year: i32, month: u8, day: u8) -> Result<Self> {
22        if !(1..=12).contains(&month) {
23            return Err(NpdatetimeError::InvalidDate(format!(
24                "Month must be between 1 and 12, got {}",
25                month
26            )));
27        }
28
29        let cal = BsCalendar::new();
30        let max_day = cal.calculate_month_days(year, month);
31        if day < 1 || day > max_day {
32            return Err(NpdatetimeError::InvalidDate(format!(
33                "Day must be between 1 and {}, got {}",
34                max_day, day
35            )));
36        }
37
38        Ok(BsDate { year, month, day })
39    }
40
41    /// Convert Julian Day to BS Date
42    pub fn from_julian_day(jd: JulianDay) -> Result<Self> {
43        // Convert to Nepal Local Time
44        let npt_jd = utc_to_npt(jd);
45        let (g_year, _g_month, _g_day, _) = npt_jd.to_gregorian();
46
47        // Approximate BS year. Most of the year, BS = G + 57.
48        // Baisakh usually starts in April (4).
49        let mut bs_year = g_year + 57;
50        let cal = BsCalendar::new();
51
52        use crate::astronomical::solar::sankranti::SankrantiFinder;
53        // Search for Mesh Sankranti in the current Gregorian year
54        let mesh_sankranti =
55            SankrantiFinder::find_sankranti(0, JulianDay::from_gregorian(g_year, 4, 1, 0.0))
56                .map_err(NpdatetimeError::CalculationError)?;
57
58        let mut npt_mesh_jd = utc_to_npt(mesh_sankranti.julian_day);
59
60        if npt_jd.0.floor() < npt_mesh_jd.0.floor() {
61            bs_year -= 1;
62            let prev_mesh = SankrantiFinder::find_sankranti(
63                0,
64                JulianDay::from_gregorian(g_year - 1, 4, 1, 0.0),
65            )
66            .map_err(NpdatetimeError::CalculationError)?;
67            npt_mesh_jd = utc_to_npt(prev_mesh.julian_day);
68        }
69
70        let mut remaining_days = (npt_jd.0.floor() - npt_mesh_jd.0.floor()) as i64;
71        let mut bs_month = 1u8;
72
73        let info = cal
74            .get_year_info(bs_year)
75            .map_err(NpdatetimeError::CalculationError)?;
76
77        while bs_month <= 12 {
78            let month_days = info.month_lengths[bs_month as usize - 1] as i64;
79            if remaining_days >= month_days {
80                remaining_days -= month_days;
81                bs_month += 1;
82            } else {
83                break;
84            }
85        }
86
87        if bs_month > 12 {
88            // This case should theoretically not be hit if month_lengths are correct
89            bs_year += 1;
90            bs_month = 1;
91        }
92
93        Ok(BsDate {
94            year: bs_year,
95            month: bs_month,
96            day: (remaining_days + 1) as u8,
97        })
98    }

100    /// Convert BS Date to Julian Day (approximate to start of day in NPT)
101    pub fn to_julian_day(&self) -> Result<JulianDay> {
102        use crate::astronomical::solar::sankranti::SankrantiFinder;
103
104        // CONSISTENCY: We use the same anchor logic as from_julian_day
105        let mesh_sankranti = SankrantiFinder::find_sankranti(
106            0,
107            JulianDay::from_gregorian(self.year - 57, 4, 1, 0.0),
108        )
109        .map_err(NpdatetimeError::CalculationError)?;
110
111        let npt_mesh_jd = utc_to_npt(mesh_sankranti.julian_day);
112        let mut total_days = 0i64;
113
114        let cal = BsCalendar::new();
115        let info = cal
116            .get_year_info(self.year)
117            .map_err(NpdatetimeError::CalculationError)?;
118
119        for m in 1..self.month {
120            total_days += info.month_lengths[m as usize - 1] as i64;
121        }
122
123        total_days += (self.day - 1) as i64;
124
125        // Start of the day in NPT
126        let jd_npt = JulianDay(npt_mesh_jd.0.floor() + total_days as f64 + 0.5); // Use midday for better round-tripping
127
128        // Convert back to UTC
129        use crate::astronomical::core::time::npt_to_utc;
130        Ok(npt_to_utc(jd_npt))
131    }
132
133    pub fn to_gregorian(&self) -> Result<(i32, u8, u8)> {
134        let jd = self.to_julian_day()?;
135        let (y, m, d, _) = jd.to_gregorian();
136        Ok((y, m, d))
137    }
138
139    pub fn from_gregorian(year: i32, month: u8, day: u8) -> Result<Self> {
140        let jd = JulianDay::from_gregorian(year, month, day, 12.0); // Midday
141        Self::from_julian_day(jd)
142    }
143}
144
145impl fmt::Display for BsDate {
146    fn fmt(&self, f: &mut fmt::Formatter<'_>) -> fmt::Result {
147        write!(f, "{}-{:02}-{:02}", self.year, self.month, self.day)
148    }
149}

151#[cfg(test)]
152mod tests {
153    use super::*;
154
155    #[test]
156    fn test_bs_date_creation() {
157        let date = BsDate::new(2081, 1, 1).unwrap();
158        assert_eq!(date.year, 2081);
159        assert_eq!(date.month, 1);
160        assert_eq!(date.day, 1);
161    }
162
163    #[test]
164    fn test_conversion_round_trip() {
165        let original = BsDate::new(2081, 1, 1).unwrap();
166        let jd = original.to_julian_day().unwrap();
167        let round_trip = BsDate::from_julian_day(jd).unwrap();
168        assert_eq!(original, round_trip);
169    }
170
171    #[test]
172    fn test_specific_date_2081_baisakh_1() {
173        // 2081 Baisakh 1 is 2024-04-13
174        let date = BsDate::from_gregorian(2024, 4, 13).unwrap();
175        assert_eq!(date.year, 2081);
176        assert_eq!(date.month, 1);
177        assert_eq!(date.day, 1);
178    }
