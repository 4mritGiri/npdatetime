1//! Solar month calculator for Bikram Sambat
2//!
3//! Determines month lengths by finding the Gregorian dates of consecutive
4//! Sankrantis in Nepal Local Time (UTC+5:45).
5
6use crate::astronomical::core::time::utc_to_npt;
7use crate::astronomical::solar::sankranti::SankrantiFinder;
8
9pub struct SolarMonthCalculator;
10
11impl SolarMonthCalculator {
12    /// Calculate the lengths of all 12 months for a given BS year
13    ///
14    /// Returns a vector of 12 integers representing the number of days in each month
15    /// (Baisakh, Jestha, ..., Chaitra)
16    pub fn calculate_month_lengths(bs_year: i32) -> Result<Vec<u8>, String> {
17        // Get Sankrantis for the current year
18        let current_year_sankrantis = SankrantiFinder::find_all_in_year(bs_year)?;
19
20        // Get Mesh Sankranti of the NEXT year to find Chaitra's length
21        let next_year_mesh = SankrantiFinder::find_sankranti(
22            0,
23            current_year_sankrantis[11].julian_day.add_days(25.0),
24        )?;
25
26        let mut all_sankrantis = current_year_sankrantis;
27        all_sankrantis.push(next_year_mesh);
28
29        let mut lengths = Vec::with_capacity(12);
30
31        for i in 0..12 {
32            let start_jd = all_sankrantis[i].julian_day;
33            let end_jd = all_sankrantis[i + 1].julian_day;
34
35            // Convert JDs to Nepal Local Time and get Gregorian dates
36            let (start_y, start_m, start_d, _) = utc_to_npt(start_jd).to_gregorian();
37            let (end_y, end_m, end_d, _) = utc_to_npt(end_jd).to_gregorian();
38
39            // Calculate total days between these two Gregorian dates
40            let length =
41                Self::days_between_gregorian((start_y, start_m, start_d), (end_y, end_m, end_d));
42
43            lengths.push(length as u8);
44        }
45
46        Ok(lengths)
47    }
48
49    /// Helper to calculate days between two Gregorian dates
50    fn days_between_gregorian(start: (i32, u8, u8), end: (i32, u8, u8)) -> i64 {
51        use crate::core::date::gregorian_to_days;
52        let start_days = gregorian_to_days(start.0, start.1, start.2);
53        let end_days = gregorian_to_days(end.0, end.1, end.2);
54        end_days - start_days
55    }
56}
