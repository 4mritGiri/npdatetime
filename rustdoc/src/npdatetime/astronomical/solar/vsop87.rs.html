1//! VSOP87 simplified planetary theory for Sun position
2//!
3//! Implements a reduced-term version of VSOP87 for calculating
4//! Earth's heliocentric position and deriving Sun's geocentric position.
5//!
6//! This implementation uses the most significant terms from VSOP87D
7//! (heliocentric spherical coordinates) providing ~0.01° accuracy.
8
9use crate::astronomical::core::{JulianDay, constants::*};
10
11/// VSOP87 term: amplitude, phase, rate
12#[derive(Debug, Clone, Copy)]
13struct VsopTerm {
14    amplitude: f64,
15    phase: f64,
16    rate: f64,
17}
18
19impl VsopTerm {
20    const fn new(amplitude: f64, phase: f64, rate: f64) -> Self {
21        Self {
22            amplitude,
23            phase,
24            rate,
25        }
26    }
27
28    /// Evaluate this term at given Julian centuries
29    fn eval(&self, t: f64) -> f64 {
30        self.amplitude * (self.phase + self.rate * t).cos()
31    }
32}

102// ============================================================================
103// CALCULATION FUNCTIONS
104// ============================================================================
105
106/// Calculate Earth's heliocentric longitude (in radians)
107fn earth_heliocentric_longitude_rad(jd: JulianDay) -> f64 {
108    let t = jd.centuries_since_j2000() / 10.0; // Convert to millennia for these coefficients
109
110    let l0: f64 = L0_TERMS.iter().map(|term| term.eval(t)).sum();
111    let l1: f64 = L1_TERMS.iter().map(|term| term.eval(t)).sum();
112    let l2: f64 = L2_TERMS.iter().map(|term| term.eval(t)).sum();
113
114    l0 + l1 * t + l2 * t * t
115}
116
117/// Calculate Earth's heliocentric latitude (in radians)
118fn earth_heliocentric_latitude_rad(jd: JulianDay) -> f64 {
119    let t = jd.centuries_since_j2000() / 10.0;
120
121    let b0: f64 = B0_TERMS.iter().map(|term| term.eval(t)).sum();
122    let b1: f64 = B1_TERMS.iter().map(|term| term.eval(t)).sum();
123
124    b0 + b1 * t
125}
126
127/// Calculate Earth's radius vector (distance from Sun in AU)
128fn earth_radius_vector(jd: JulianDay) -> f64 {
129    let t = jd.centuries_since_j2000() / 10.0;
130
131    let r0: f64 = R0_TERMS.iter().map(|term| term.eval(t)).sum();
132    let r1: f64 = R1_TERMS.iter().map(|term| term.eval(t)).sum();
133
134    r0 + r1 * t
135}
136
137/// Normalize angle to [0, 360) degrees
138fn normalize_degrees(angle: f64) -> f64 {
139    angle.rem_euclid(360.0)
140}

146/// VSOP87 calculator for high-precision solar positions
147pub struct Vsop87Calculator;
148
149impl Vsop87Calculator {
150    /// Calculate Earth's heliocentric longitude (in degrees)
151    ///
152    /// This is Earth's ecliptic longitude as seen from the Sun
153    pub fn earth_heliocentric_longitude(jd: JulianDay) -> f64 {
154        let lon_rad = earth_heliocentric_longitude_rad(jd);
155        normalize_degrees(lon_rad * RAD_TO_DEG)
156    }
157
158    /// Calculate Earth's heliocentric latitude (in degrees)
159    ///
160    /// This is Earth's ecliptic latitude as seen from the Sun
161    /// (usually very small, close to 0)
162    pub fn earth_heliocentric_latitude(jd: JulianDay) -> f64 {
163        let lat_rad = earth_heliocentric_latitude_rad(jd);
164        lat_rad * RAD_TO_DEG
165    }
166
167    /// Calculate Earth-Sun distance (in AU)
168    pub fn earth_sun_distance(jd: JulianDay) -> f64 {
169        earth_radius_vector(jd)
170    }
171
172    /// Calculate Sun's geocentric longitude (in degrees)
173    ///
174    /// This is the Sun's apparent position as seen from Earth.
175    /// It's Earth's heliocentric longitude + 180°
176    pub fn sun_geocentric_longitude(jd: JulianDay) -> f64 {
177        let earth_lon = Self::earth_heliocentric_longitude(jd);
178        normalize_degrees(earth_lon + 180.0)
179    }
180
181    /// Calculate Sun's true longitude (alias for geocentric longitude)
182    ///
183    /// In Indian astronomy, this is called Sayana (tropical) longitude
184    pub fn sun_true_longitude(jd: JulianDay) -> f64 {
185        Self::sun_geocentric_longitude(jd)
186    }
187
188    /// Calculate nutation in longitude (simplified)
189    ///
190    /// Returns correction to longitude in degrees
191    fn nutation_longitude(jd: JulianDay) -> f64 {
192        let t = jd.centuries_since_j2000();
193
194        // Mean longitude of lunar ascending node
195        let omega = 125.04 - 1934.136 * t;
196        let omega_rad = omega * DEG_TO_RAD;
197
198        // Simplified nutation formula
199        -0.00569 - 0.00478 * omega_rad.sin()
200    }
201
202    /// Calculate Sun's apparent longitude (includes nutation and aberration)
203    ///
204    /// This is the most accurate representation of Sun's position
205    pub fn sun_apparent_longitude(jd: JulianDay) -> f64 {
206        let true_lon = Self::sun_true_longitude(jd);
207        let nutation_and_aberration = Self::nutation_longitude(jd);
208
209        normalize_degrees(true_lon + nutation_and_aberration)
210    }
211}

213#[cfg(test)]
214mod tests {
215    use super::*;
216
217    #[test]
218    fn test_earth_longitude_j2000() {
219        let jd = JulianDay(J2000_0);
220        let lon = Vsop87Calculator::earth_heliocentric_longitude(jd);
221
222        // At J2000.0, Earth should be near 100° heliocentric longitude
223        assert!(
224            (lon - 100.46).abs() < 1.0,
225            "Earth longitude at J2000.0 = {}",
226            lon
227        );
228    }
229
230    #[test]
231    fn test_sun_longitude_j2000() {
232        let jd = JulianDay(J2000_0);
233        let lon = Vsop87Calculator::sun_geocentric_longitude(jd);
234
235        // Sun should be at Earth_lon + 180°, so around 280°
236        assert!(
237            (lon - 280.46).abs() < 1.0,
238            "Sun longitude at J2000.0 = {}",
239            lon
240        );
241    }
242
243    #[test]
244    fn test_earth_sun_distance() {
245        let jd = JulianDay(J2000_0);
246        let dist = Vsop87Calculator::earth_sun_distance(jd);
247
248        // Distance should be close to 1 AU
249        assert!(
250            (dist - 1.0).abs() < 0.02,
251            "Earth-Sun distance = {} AU",
252            dist
253        );
254    }
255
256    #[test]
257    fn test_earth_latitude_near_zero() {
258        let jd = JulianDay(J2000_0);
259        let lat = Vsop87Calculator::earth_heliocentric_latitude(jd);
260
261        // Earth's latitude should be very small (< 0.001°)
262        assert!(lat.abs() < 0.001, "Earth latitude = {}°", lat);
263    }
264
265    #[test]
266    fn test_vernal_equinox_2020() {
267        // March 20, 2020, 03:50 UTC (vernal equinox)
268        let jd = JulianDay::from_gregorian(2020, 3, 20, 3.833);
269        let lon = Vsop87Calculator::sun_apparent_longitude(jd);
270
271        // Sun should be very close to 0° at vernal equinox
272        let diff = if lon > 180.0 { 360.0 - lon } else { lon };
273        assert!(diff < 0.5, "Sun longitude at vernal equinox = {}°", lon);
274    }
