1//! ELP-2000 simplified lunar theory for Moon position
2//!
3//! Implements a reduced-term version of ELP-2000 for calculating
4//! geocentric position of the Moon.
5//!
6//! This implementation uses the fundamental arguments and most significant
7//! periodic terms to provide accuracy suitable for Tithi and eclipse calculations.
8
9use crate::astronomical::core::{JulianDay, constants::*};

11/// Multipliers for fundamental arguments (D, M, M', F)
12#[derive(Debug, Clone, Copy)]
13struct LunarTerm {
14    d: i8,
15    m: i8,
16    m_prime: i8,
17    f: i8,
18    amplitude: f64,
19}
20
21impl LunarTerm {
22    const fn new(d: i8, m: i8, m_prime: i8, f: i8, amplitude: f64) -> Self {
23        Self {
24            d,
25            m,
26            m_prime,
27            f,
28            amplitude,
29        }
30    }
31}

74/// Fundamental arguments of Moon's motion
75#[derive(Debug, Clone, Copy)]
76struct FundamentalArgs {
77    l_prime: f64, // Mean longitude of Moon
78    d: f64,       // Mean elongation of Moon (Moon - Sun)
79    m: f64,       // Mean anomaly of Sun
80    m_prime: f64, // Mean anomaly of Moon
81    f: f64,       // Mean distance of Moon from its ascending node
82}
83
84impl FundamentalArgs {
85    fn calculate(jd: JulianDay) -> Self {
86        let t = jd.centuries_since_j2000();
87        let t2 = t * t;
88        let t3 = t * t * t;
89        let t4 = t * t * t * t;
90
91        // Meeus, Formulas 47.1 - 47.5
92        let l_prime =
93            218.3164477 + 481267.88123421 * t - 0.0015786 * t2 + t3 / 538841.0 - t4 / 65194000.0;
94        let d =
95            297.8501921 + 445267.1114034 * t - 0.0018819 * t2 + t3 / 545868.0 - t4 / 113065000.0;
96        let m = 357.5291092 + 35999.0502909 * t - 0.0001536 * t2 + t3 / 24490000.0;
97        let m_prime =
98            134.9633964 + 477198.8675055 * t + 0.0087414 * t2 + t3 / 69699.0 - t4 / 14712000.0;
99        let f =
100            93.2720950 + 483202.0175233 * t - 0.0036539 * t2 - t3 / 3526000.0 + t4 / 863310000.0;
101
102        Self {
103            l_prime: l_prime.rem_euclid(360.0),
104            d: d.rem_euclid(360.0),
105            m: m.rem_euclid(360.0),
106            m_prime: m_prime.rem_euclid(360.0),
107            f: f.rem_euclid(360.0),
108        }
109    }
110}

112pub struct Elp2000Calculator;
113
114impl Elp2000Calculator {
115    /// Calculate Moon's geocentric longitude (in degrees)
116    pub fn geocentric_longitude(jd: JulianDay) -> f64 {
117        let args = FundamentalArgs::calculate(jd);
118        let t = jd.centuries_since_j2000();
119        let e = 1.0 - 0.002516 * t - 0.0000074 * t * t;
120
121        let mut delta_l = 0.0;
122        for term in LONG_TERMS {
123            let arg = (term.d as f64 * args.d
124                + term.m as f64 * args.m
125                + term.m_prime as f64 * args.m_prime
126                + term.f as f64 * args.f)
127                * DEG_TO_RAD;
128
129            let mut coeff = term.amplitude;
130            // Correction for Sun's eccentricity (Meeus formula 47)
131            if term.m == 1 || term.m == -1 {
132                coeff *= e;
133            } else if term.m == 2 || term.m == -2 {
134                coeff *= e * e;
135            }
136
137            delta_l += coeff * arg.sin();
138        }
139
140        // Add additional corrections from planetary perturbations (Meeus p. 341)
141        let a1 = (119.75 + 131.849 * t).rem_euclid(360.0) * DEG_TO_RAD;
142        let a2 = (53.09 + 479264.290 * t).rem_euclid(360.0) * DEG_TO_RAD;
143        let a3 = (313.45 + 481266.484 * t).rem_euclid(360.0) * DEG_TO_RAD;
144
145        delta_l += 39.5 * a1.sin();
146        delta_l += 31.8 * a2.sin();
147        delta_l += 19.6 * a3.sin();
148
149        (args.l_prime + delta_l / 10000.0).rem_euclid(360.0)
150    }
151
152    /// Calculate Moon's distance from Earth (in km)
153    pub fn distance(jd: JulianDay) -> f64 {
154        let args = FundamentalArgs::calculate(jd);
155        let t = jd.centuries_since_j2000();
156        let e = 1.0 - 0.002516 * t - 0.0000074 * t * t;
157
158        let mut delta_r = 0.0;
159        for term in DIST_TERMS {
160            let arg = (term.d as f64 * args.d
161                + term.m as f64 * args.m
162                + term.m_prime as f64 * args.m_prime
163                + term.f as f64 * args.f)
164                * DEG_TO_RAD;
165
166            let mut coeff = term.amplitude;
167            if term.m == 1 || term.m == -1 {
168                coeff *= e;
169            } else if term.m == 2 || term.m == -2 {
170                coeff *= e * e;
171            }
172
173            delta_r += coeff * arg.cos();
174        }
175
176        385000.56 + delta_r
177    }
178
179    /// Calculate Moon's apparent longitude (includes nutation)
180    pub fn apparent_longitude(jd: JulianDay) -> f64 {
181        let geo_long = Self::geocentric_longitude(jd);
182
183        // Nutation in longitude (simplified, same as solar)
184        let t = jd.centuries_since_j2000();
185        let omega = 125.04 - 1934.136 * t;
186        let omega_rad = omega * DEG_TO_RAD;
187        let nutation = -0.00569 - 0.00478 * omega_rad.sin();
188
189        (geo_long + nutation).rem_euclid(360.0)
190    }
191}
